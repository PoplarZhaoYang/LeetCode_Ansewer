res 是局部变量，函数结束，内存被回收，所以使用 malloc 分配内存

```
int* twoSum(int* nums, int numsSize, int target) {
	int i, j;
	int *res = (int* )malloc(sizeof(int) * 2);

	for (i=0;i<numsSize;i++) {
		for(j=i+1;j<numsSize;j++) {
			if ((nums[i] + nums[j]) == target) {
				res[0] = i;
				res[1] = j;
				break;
			}
		}
	}
	return res;
}
```
